create table Restaurant(
id INT primary key,
rate INT not null,
name VARCHAR(255) not null,
CONSTRAINT rate_check CHECK(rate > 0)
);

create table Employee(
id INT primary key,
salary INT,
name VARCHAR(255) NOT NULL,
RestaurantId INT,
CONSTRAINT salary_check CHECK(salary > 0),
constraint FK_RESTAURANT_ID foreign key(RestaurantId) references Restaurant(id)
);

create table Guest(
id int primary key,
Tip int not null,
EmployeId INT,
constraint FK_EMPLOYEE_ID foreign key(EmployeId) references Employee(id)
);

create table Food(
id int primary key,
name VARCHAR(255),
GuestId int,
constraint FK_GUEST foreign key (GuestId) references Guest(id)
);


create table reservation(
broj int primary key,
people int not null,
GuestId int,
constraint FK_GUEST_reservation foreign key (GuestId) references Guest(id)
);
